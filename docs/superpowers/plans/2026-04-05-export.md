**Goal:** Add Export Table (multi-format) and Export Database (SQL dump) dialogs accessible from sidebar context menus.

**Tech Stack:** React, TypeScript, Electron IPC, Electron `dialog.showSaveDialog`, mysql2

```typescript
// src/renderer/utils/format-rows.ts

export function cellToString(val: unknown): string {
  return val === null || val === undefined ? 'NULL' : String(val);
}

export function rowsToTsv(rows: Record<string, unknown>[], colNames: string[]): string {
  return colNames.join('\t') + '\n' + rows.map(r => colNames.map(c => cellToString(r[c])).join('\t')).join('\n');
}

export function rowsToCsv(rows: Record<string, unknown>[], colNames: string[]): string {
  const esc = (v: string) => v.includes(',') || v.includes('"') || v.includes('\n') ? `"${v.replace(/"/g, '""')}"` : v;
  return colNames.map(esc).join(',') + '\n' + rows.map(r => colNames.map(c => esc(cellToString(r[c]))).join(',')).join('\n');
}

export function rowsToMarkdown(rows: Record<string, unknown>[], colNames: string[]): string {
  return '| ' + colNames.join(' | ') + ' |\n| ' + colNames.map(() => '---').join(' | ') + ' |\n' +
    rows.map(r => '| ' + colNames.map(c => cellToString(r[c])).join(' | ') + ' |').join('\n');
}

export function rowsToJson(rows: Record<string, unknown>[], colNames: string[]): string {
  return JSON.stringify(rows.map(r => { const o: any = {}; colNames.forEach(c => o[c] = r[c]); return o; }), null, 2);
}

export function rowsToSqlInsert(rows: Record<string, unknown>[], colNames: string[], tableName: string): string {
  return rows.map(r => {
    const vals = colNames.map(c => {
      const v = r[c];
      return v === null || v === undefined ? 'NULL' : typeof v === 'number' ? String(v) : `'${String(v).replace(/'/g, "\\'")}'`;
    }).join(', ');
    return `INSERT INTO \`${tableName}\` (${colNames.map(c => '`' + c + '`').join(', ')}) VALUES (${vals});`;
  }).join('\n');
}

export function rowsToHtml(rows: Record<string, unknown>[], colNames: string[]): string {
  return `<table>\n<thead>\n<tr>${colNames.map(c => `<th>${c}</th>`).join('')}</tr>\n</thead>\n<tbody>\n` +
    rows.map(r => '<tr>' + colNames.map(c => `<td>${cellToString(r[c])}</td>`).join('') + '</tr>').join('\n') +
    '\n</tbody>\n</table>';
}

export type ExportFormat = 'csv' | 'tsv' | 'json' | 'sql' | 'markdown' | 'html';

export const FORMAT_LABELS: Record<ExportFormat, string> = {
  sql: 'SQL', csv: 'CSV', json: 'JSON', markdown: 'Markdown', tsv: 'TSV', html: 'HTML Table',
};

export const FORMAT_EXT: Record<ExportFormat, string> = {
  sql: 'sql', csv: 'csv', json: 'json', markdown: 'md', tsv: 'tsv', html: 'html',
};

export function formatRows(rows: Record<string, unknown>[], colNames: string[], format: ExportFormat, tableName: string): string {
  switch (format) {
    case 'tsv': return rowsToTsv(rows, colNames);
    case 'csv': return rowsToCsv(rows, colNames);
    case 'markdown': return rowsToMarkdown(rows, colNames);
    case 'json': return rowsToJson(rows, colNames);
    case 'sql': return rowsToSqlInsert(rows, colNames, tableName);
    case 'html': return rowsToHtml(rows, colNames);
  }
}
```

```typescript
import { useState, useEffect } from 'react';
import { useIpc } from '../hooks/use-ipc';
import { ColumnMeta } from '../../shared/types';
import { ExportFormat, FORMAT_LABELS, FORMAT_EXT, formatRows, rowsToSqlInsert } from '../utils/format-rows';

interface Props {
  connectionId: string;
  database: string;
  table: string;
  onClose: () => void;
}

const FORMATS: ExportFormat[] = ['sql', 'csv', 'json', 'markdown', 'tsv', 'html'];

export default function ExportTableDialog({ connectionId, database, table, onClose }: Props) {
  const ipc = useIpc();
  const [format, setFormat] = useState<ExportFormat>('sql');
  const [fileName, setFileName] = useState(`${database}.${table}.sql`);
  const [createTable, setCreateTable] = useState(true);
  const [ifNotExists, setIfNotExists] = useState(true);
  const [exportData, setExportData] = useState(true);
  const [columns, setColumns] = useState<ColumnMeta[]>([]);
  const [selectedFields, setSelectedFields] = useState<Set<string>>(new Set());
  const [exporting, setExporting] = useState(false);
  const [error, setError] = useState('');

  const isSql = format === 'sql';

  useEffect(() => {
    ipc.schemaDescribe(connectionId, database, table).then((cols: ColumnMeta[]) => {
      setColumns(cols);
      setSelectedFields(new Set(cols.map(c => c.name)));
    });
  }, [connectionId, database, table]);

  const handleFormatChange = (f: ExportFormat) => {
    setFormat(f);
    setFileName(`${database}.${table}.${FORMAT_EXT[f]}`);
  };

  const allSelected = columns.length > 0 && selectedFields.size === columns.length;
  const toggleAll = () => {
    if (allSelected) {
      setSelectedFields(new Set());
    } else {
      setSelectedFields(new Set(columns.map(c => c.name)));
    }
  };
  const toggleField = (name: string) => {
    setSelectedFields(prev => {
      const next = new Set(prev);
      if (next.has(name)) next.delete(name); else next.add(name);
      return next;
    });
  };

  const handleExport = async () => {
    setExporting(true);
    setError('');
    try {
      const filePath = await ipc.exportPickSaveFile(fileName, FORMAT_EXT[format]);
      if (!filePath) { setExporting(false); return; }

      let content = '';

      if (isSql && createTable) {
        let ddl: string = await ipc.schemaCreateTable(connectionId, database, table);
        if (ifNotExists) {
          ddl = ddl.replace(/^CREATE TABLE/i, 'CREATE TABLE IF NOT EXISTS');
        }
        content += ddl + ';\n\n';
      }

      const shouldExportData = isSql ? exportData : true;
      if (shouldExportData && selectedFields.size > 0) {
        const colNames = columns.map(c => c.name).filter(c => selectedFields.has(c));
        const rows = await ipc.exportFetchAllRows(connectionId, database, table, colNames);

        if (isSql) {
          content += rowsToSqlInsert(rows, colNames, table);
        } else {
          content = formatRows(rows, colNames, format, table);
        }
      }

      await ipc.exportWriteFile(filePath, content);
      onClose();
    } catch (err: any) {
      setError(err.message || 'Export failed');
    } finally {
      setExporting(false);
    }
  };

  const fieldListDisabled = isSql && !exportData;

  return (
    <div className="modal-overlay" onClick={onClose}>
      <div className="modal" onClick={e => e.stopPropagation()} style={{ minWidth: 480 }}>
        <div className="modal-title">Export {database}.{table}</div>

        <label style={{ display: 'block', marginBottom: 12 }}>
          <div style={{ fontSize: 11, color: '#888', marginBottom: 4 }}>Format</div>
          <div style={{ display: 'flex', gap: 4, flexWrap: 'wrap' }}>
            {FORMATS.map(f => (
              <button key={f} className={`btn ${format === f ? 'btn-primary' : 'btn-secondary'}`}
                onClick={() => handleFormatChange(f)}>{FORMAT_LABELS[f]}</button>
            ))}
          </div>
        </label>

        <label style={{ display: 'block', marginBottom: 12 }}>
          <div style={{ fontSize: 11, color: '#888', marginBottom: 4 }}>File Name</div>
          <input className="input" value={fileName} onChange={e => setFileName(e.target.value)} />
        </label>

        {isSql && (
          <div style={{ marginBottom: 12 }}>
            <label style={{ display: 'flex', alignItems: 'center', gap: 6, fontSize: 12, cursor: 'pointer' }}>
              <input type="checkbox" checked={createTable} onChange={e => setCreateTable(e.target.checked)} />
              CREATE TABLE statement
            </label>
            {createTable && (
              <label style={{ display: 'flex', alignItems: 'center', gap: 6, fontSize: 12, cursor: 'pointer', marginLeft: 20, marginTop: 4 }}>
                <input type="checkbox" checked={ifNotExists} onChange={e => setIfNotExists(e.target.checked)} />
                IF NOT EXISTS
              </label>
            )}
          </div>
        )}

        <div style={{ marginBottom: 12 }}>
          <label style={{ display: 'flex', alignItems: 'center', gap: 6, fontSize: 12, cursor: isSql ? 'pointer' : 'default', opacity: isSql ? 1 : 0.5 }}>
            <input type="checkbox" checked={isSql ? exportData : true} disabled={!isSql}
              onChange={e => setExportData(e.target.checked)} />
            Export Data
          </label>
        </div>

        <div style={{ marginBottom: 12, opacity: fieldListDisabled ? 0.4 : 1 }}>
          <div style={{ fontSize: 11, color: '#888', marginBottom: 4, display: 'flex', alignItems: 'center', gap: 8 }}>
            Fields
            <span style={{ cursor: fieldListDisabled ? 'default' : 'pointer', color: '#4b6eaf', fontSize: 10 }}
              onClick={() => !fieldListDisabled && toggleAll()}>
              {allSelected ? 'Deselect All' : 'Select All'}
            </span>
          </div>
          <div style={{ maxHeight: 150, overflowY: 'auto', border: '1px solid #515151', borderRadius: 4, padding: 4 }}>
            {columns.map(c => (
              <label key={c.name} style={{ display: 'flex', alignItems: 'center', gap: 6, fontSize: 12, padding: '2px 4px', cursor: fieldListDisabled ? 'default' : 'pointer' }}>
                <input type="checkbox" checked={selectedFields.has(c.name)} disabled={fieldListDisabled}
                  onChange={() => toggleField(c.name)} />
                {c.name} <span style={{ color: '#666', fontSize: 10 }}>{c.type}</span>
              </label>
            ))}
          </div>
        </div>

        {error && (
          <div style={{ padding: 8, borderRadius: 4, marginBottom: 12, fontSize: 12, background: '#2a0000', color: '#c75450' }}>
            {error}
          </div>
        )}

        <div style={{ display: 'flex', gap: 8, justifyContent: 'flex-end' }}>
          <button className="btn btn-secondary" onClick={onClose}>Cancel</button>
          <button className="btn btn-primary" onClick={handleExport} disabled={exporting}>
            {exporting ? 'Exporting...' : 'Export'}
          </button>
        </div>
      </div>
    </div>
  );
}
```

```typescript
import { useState, useEffect } from 'react';
import { useIpc } from '../hooks/use-ipc';
import { rowsToSqlInsert } from '../utils/format-rows';

interface Props {
  connectionId: string;
  database: string;
  onClose: () => void;
}

export default function ExportDatabaseDialog({ connectionId, database, onClose }: Props) {
  const ipc = useIpc();
  const [fileName, setFileName] = useState(`${database}.sql`);
  const [createDatabase, setCreateDatabase] = useState(true);
  const [ifNotExistsDb, setIfNotExistsDb] = useState(true);
  const [createTable, setCreateTable] = useState(true);
  const [ifNotExistsTbl, setIfNotExistsTbl] = useState(true);
  const [exportData, setExportData] = useState(true);
  const [tables, setTables] = useState<string[]>([]);
  const [selectedTables, setSelectedTables] = useState<Set<string>>(new Set());
  const [exporting, setExporting] = useState(false);
  const [error, setError] = useState('');

  useEffect(() => {
    ipc.schemaTables(connectionId, database).then((tbls: string[]) => {
      setTables(tbls);
      setSelectedTables(new Set(tbls));
    });
  }, [connectionId, database]);

  const allSelected = tables.length > 0 && selectedTables.size === tables.length;
  const toggleAll = () => {
    if (allSelected) {
      setSelectedTables(new Set());
    } else {
      setSelectedTables(new Set(tables));
    }
  };
  const toggleTable = (name: string) => {
    setSelectedTables(prev => {
      const next = new Set(prev);
      if (next.has(name)) next.delete(name); else next.add(name);
      return next;
    });
  };

  const handleExport = async () => {
    if (selectedTables.size === 0) return;
    setExporting(true);
    setError('');
    try {
      const filePath = await ipc.exportPickSaveFile(fileName, 'sql');
      if (!filePath) { setExporting(false); return; }

      const parts: string[] = [];

      if (createDatabase) {
        const stmt = ifNotExistsDb
          ? `CREATE DATABASE IF NOT EXISTS \`${database}\`;`
          : `CREATE DATABASE \`${database}\`;`;
        parts.push(stmt);
      }

      parts.push(`USE \`${database}\`;`);
      parts.push('');

      for (const table of tables) {
        if (!selectedTables.has(table)) continue;

        if (createTable) {
          let ddl: string = await ipc.schemaCreateTable(connectionId, database, table);
          if (ifNotExistsTbl) {
            ddl = ddl.replace(/^CREATE TABLE/i, 'CREATE TABLE IF NOT EXISTS');
          }
          parts.push(ddl + ';');
          parts.push('');
        }

        if (exportData) {
          const rows: Record<string, unknown>[] = await ipc.exportFetchAllRows(connectionId, database, table);
          if (rows.length > 0) {
            const colNames = Object.keys(rows[0]);
            parts.push(rowsToSqlInsert(rows, colNames, table));
            parts.push('');
          }
        }
      }

      await ipc.exportWriteFile(filePath, parts.join('\n'));
      onClose();
    } catch (err: any) {
      setError(err.message || 'Export failed');
    } finally {
      setExporting(false);
    }
  };

  return (
    <div className="modal-overlay" onClick={onClose}>
      <div className="modal" onClick={e => e.stopPropagation()} style={{ minWidth: 480 }}>
        <div className="modal-title">Export Database: {database}</div>

        <label style={{ display: 'block', marginBottom: 12 }}>
          <div style={{ fontSize: 11, color: '#888', marginBottom: 4 }}>File Name</div>
          <input className="input" value={fileName} onChange={e => setFileName(e.target.value)} />
        </label>

        <div style={{ marginBottom: 12 }}>
          <label style={{ display: 'flex', alignItems: 'center', gap: 6, fontSize: 12, cursor: 'pointer' }}>
            <input type="checkbox" checked={createDatabase} onChange={e => setCreateDatabase(e.target.checked)} />
            CREATE DATABASE statement
          </label>
          {createDatabase && (
            <label style={{ display: 'flex', alignItems: 'center', gap: 6, fontSize: 12, cursor: 'pointer', marginLeft: 20, marginTop: 4 }}>
              <input type="checkbox" checked={ifNotExistsDb} onChange={e => setIfNotExistsDb(e.target.checked)} />
              IF NOT EXISTS
            </label>
          )}
        </div>

        <div style={{ marginBottom: 12 }}>
          <label style={{ display: 'flex', alignItems: 'center', gap: 6, fontSize: 12, cursor: 'pointer' }}>
            <input type="checkbox" checked={createTable} onChange={e => setCreateTable(e.target.checked)} />
            CREATE TABLE statements
          </label>
          {createTable && (
            <label style={{ display: 'flex', alignItems: 'center', gap: 6, fontSize: 12, cursor: 'pointer', marginLeft: 20, marginTop: 4 }}>
              <input type="checkbox" checked={ifNotExistsTbl} onChange={e => setIfNotExistsTbl(e.target.checked)} />
              IF NOT EXISTS
            </label>
          )}
        </div>

        <div style={{ marginBottom: 12 }}>
          <label style={{ display: 'flex', alignItems: 'center', gap: 6, fontSize: 12, cursor: 'pointer' }}>
            <input type="checkbox" checked={exportData} onChange={e => setExportData(e.target.checked)} />
            Export Data
          </label>
        </div>

        <div style={{ marginBottom: 12 }}>
          <div style={{ fontSize: 11, color: '#888', marginBottom: 4, display: 'flex', alignItems: 'center', gap: 8 }}>
            Tables
            <span style={{ cursor: 'pointer', color: '#4b6eaf', fontSize: 10 }} onClick={toggleAll}>
              {allSelected ? 'Deselect All' : 'Select All'}

          </div>

}
```
